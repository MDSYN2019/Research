#+begin_src yaml 
# The workflow is named 'bump version

name: Bump version
# triggers: it is triggered on a push event to the 'develop' branch. This means
# whenever someone pushes code changes to the 'develop' branch of the repository,
# this workflow will be triggered 
on:
  push:
    branches:
      - develop
# the job runs on a virtual machine with ubuntu-latest operating system 
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
    
      - uses: actions/checkout@v2 # checks the code from the repository 
      - name: Bump version and push tag # uses action to bump the version and add a new tag 
        id: tag_version
        uses: mathieudutour/github-tag-action@v6.1
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
      - name: Create a GitHub release # creates a github release. It specifies the tag for the
	# release as the output of the previous step 
        uses: ncipollo/release-action@v1
        with:
          tag: ${{ steps.tag_version.outputs.new_tag }}
          name: Release ${{ steps.tag_version.outputs.new_tag }} # sets the release name as
	  # release followed by a new tag, and uses the changelog generated by the 'github-tag-action'
	  # as the release body
          body: ${{ steps.tag_version.outputs.changelog }}
#+end_src yaml
