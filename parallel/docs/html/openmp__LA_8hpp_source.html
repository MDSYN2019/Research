116 
117  /*
118  The box-muller algorithm - designed to convert two uniform random vairables
119  into a standard Gaussian random variable.
120 
121  Box-Muller is a good choice for random number generator if your compiler
122  does not support the C++11 standard.
123 
124  */
125 
126  double x = 0.0;
127  double y = 0.0;
128  double euclid_sq = 0.0;
129 
130  /*
131 
132  Continue genereating two uniform random variables
133  until the square of their 'euclidean distance'
134  is less than unity
135 
136  */
137 
138  // What does static cast do?
139 
140  /*
141  The static_cast operator converts variable j to type float. This allows
142  the compiler to generate a division with an answer of type float. All
143  static_cast resolve at compile time nd do not remove any const or
144  volatile modifiers
145  */
146  do {
147  x = 2.0 * rand() / static_cast<double>(RAND_MAX) -1;
148  y = 2.0 * rand() / static_cast<double>(RAND_MAX) -1;
149  euclid_sq = x*x + y*y;
150  } while (euclid_sq >= 1.0);
151 
152  return x*sqrt(-2 * log(euclid_sq)/euclid_sq);
153 }
